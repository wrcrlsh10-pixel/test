<!-- Firebase SDK -->
<script type="module">
    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
    import { getFirestore, collection, addDoc, onSnapshot, query, orderBy, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js";

    // TODO: 아래 설정을 본인의 Firebase 프로젝트 설정으로 교체하세요.
    const firebaseConfig = {
        apiKey: "YOUR_API_KEY",
        authDomain: "YOUR_PROJECT_ID.firebaseapp.com",
        projectId: "YOUR_PROJECT_ID",
        storageBucket: "YOUR_PROJECT_ID.appspot.com",
        messagingSenderId: "YOUR_SENDER_ID",
        appId: "YOUR_APP_ID"
    };

    // Firebase 초기화
    const app = initializeApp(firebaseConfig);
    const db = getFirestore(app);
    const attendanceCol = collection(db, "attendance");

    // 출석 기록 저장 함수
    window.recordAttendance = async (status) => {
        const nameInput = document.getElementById('studentName');
        const name = nameInput.value.trim();

        if (!name) {
            alert("학생 이름을 입력해주세요.");
            return;
        }

        try {
            await addDoc(attendanceCol, {
                name: name,
                status: status,
                timestamp: serverTimestamp()
            });
            nameInput.value = ""; // 입력창 초기화
        } catch (e) {
            console.error("에러 발생: ", e);
            alert("데이터 저장 중 오류가 발생했습니다.");
        }
    };

    // 실시간 데이터 감시 및 화면 표시
    const q = query(attendanceCol, orderBy("timestamp", "desc"));
    onSnapshot(q, (snapshot) => {
        const listElement = document.getElementById('attendanceList');
        const countText = document.getElementById('countText');
        listElement.innerHTML = "";
        
        countText.innerText = `기록: ${snapshot.size}건`;

        snapshot.forEach((doc) => {
            const data = doc.data();
            const time = data.timestamp ? new Date(data.timestamp.seconds * 1000).toLocaleString() : "처리 중...";
            
            let badgeClass = "badge-present";
            if(data.status === "결석") badgeClass = "badge-absent";
            if(data.status === "지각") badgeClass = "badge-tardy";

            const li = document.createElement('li');
            li.className = 'record-item';
            li.innerHTML = `
                <div class="student-info">
                    <span class="name">${data.name}</span>
                    <span class="time">${time}</span>
                </div>
                <span class="badge ${badgeClass}">${data.status}</span>
            `;
            listElement.appendChild(li);
        });
    });
</script>
